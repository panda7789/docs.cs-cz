```csharp  
   public static implicit operator int(iii x)

   }

   {  
      return null;  
   }  
  
   public static int operator ++(int aa)   // CS0559

   }

```  
  
## <a name="example"></a><span data-ttu-id="a3fe2-109">Příklad</span><span class="sxs-lookup"><span data-stu-id="a3fe2-109">Example</span></span>  
 <span data-ttu-id="a3fe2-110">Následující ukázka generuje CS0559.</span><span class="sxs-lookup"><span data-stu-id="a3fe2-110">The following sample generates CS0559.</span></span>  
  
```csharp  
// CS0559_b.cs  
// compile with: /target:library  
public struct S  
{  
   public static S operator ++(S? s) { return new S(); }   // CS0559  
   public static S operator --(S? s) { return new S(); }   // CS0559  
}  
  
public struct T  
{  
// OK  
   public static T operator --(T t) { return new T(); }  
   public static T operator ++(T t) { return new T(); }  
  
   public static T? operator --(T? t) { return new T(); }  
   public static T? operator ++(T? t) { return new T(); }  
}  
```
