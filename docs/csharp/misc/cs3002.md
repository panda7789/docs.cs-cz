public class a  
{  
    public ushort bad()   // CS3002, public method

        ushort a;

        return a;  
    }

    {  
        ushort a;  
        a = ushort.MaxValue;  
        return a;  
    }  
  
    public static void Main()  
    {  
    }  
}
