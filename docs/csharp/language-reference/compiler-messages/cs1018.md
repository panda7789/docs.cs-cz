```csharp  
// CS1018.cs

{  
}  
  
public class a : C  
{  
    public a(int i)  
    {  
    }

    // public a () resolves by removing the colon

    // public a () : this(1) calls the assignment constructor of class a

    }

    {  
        return 1;  
    }  
}
```
