 <span data-ttu-id="4e91d-106">Následující ukázka generuje CS0675:</span><span class="sxs-lookup"><span data-stu-id="4e91d-106">The following sample generates CS0675:</span></span>  
  
```csharp  
// CS0675.cs  
// compile with: /W:3  
using System;  
  
public class sign  
{  
   public static void Main()  
   {  
      int hi = 1;  
      int lo = 1;  
      long value = (((long)hi) << 32) | lo;              // CS0675  
      // try the following line instead  
      // long value = (((long)hi) << 32) | ((uint)lo);   // correct  
   }  
}  
```
